CREATE TABLE `Object` (
	`ID` int(11) PRIMARY KEY AUTO_INCREMENT,
	`Name` varchar(32) UNIQUE NOT NULL,
	`Description` varchar(128) NOT NULL
);

CREATE TABLE `Image` (
	`ID` int(11) PRIMARY KEY AUTO_INCREMENT,
	`Name` varchar(32) UNIQUE NOT NULL,
	`Description` varchar(128) NOT NULL,
	`File_name` varchar(32) UNIQUE NOT NULL
);

CREATE TABLE `Relation` (
	`Object1_ID` int(11),
	`Object2_ID` int(11),
	`Relation_description` varchar(128),
	
	CONSTRAINT `Object1_ID_FK` FOREIGN KEY (`Object1_ID`) REFERENCES Object(`ID`),
	CONSTRAINT `Object2_ID_FK` FOREIGN KEY (`Object2_ID`) REFERENCES Object(`ID`),

	PRIMARY KEY(`Object1_ID`, `Object2_ID`, `Relation_description`)
);

CREATE TABLE `Object_image` (
	`Object_ID` int(11),
	`Image_ID` int(11),
	
	CONSTRAINT `Object_ID_FK` FOREIGN KEY (`Object_ID`) REFERENCES Object(`ID`),
	CONSTRAINT `Image_ID_FK` FOREIGN KEY (`Image_ID`) REFERENCES Image(`ID`),

	PRIMARY KEY(`Object_ID`, `Image_ID`)
);
